\makeglossaries

\newglossaryentry{IEEE}{
name={IEEE},
description={\emph{Institute of Electrical and Electronics Engineers} - Institut des ingénieurs électriciens et électroniciens, association de professionnels pour le développement de technologies }}

\newglossaryentry{WiFi}{
name={WiFi},
description={\emph{Wireless Fidelity} - Fidélité sans-fil, ensemble de protocoles de communications}}

\newglossaryentry{reseau}{
name={Réseau informatique},
description={Ensemble d'équipements reliés entre eux, permettant de faire circuler des informations, une communication, etc...}}

\newglossaryentry{encryptage}{
name={Encryptage},
description={Chiffrement, procédé de cryptographie empêchant la lecture de toute personne n'ayant pas la clé de déchiffrement}}

\newglossaryentry{WEP}{
name={WEP},
description={\emph{Wired Equivalent Privacy} - Confidentialité équivalente à celle sur une connexion filaire, un encryptage d'un réseau sans-fil}}

\newglossaryentry{borne}{
name={Borne WiFi},
description={Matériel donnant l'accès à un réseau sans-fil}}

\newglossaryentry{hotspot}{
name={Hotspot},
description={Nom anglais pour un point d'accès sans-fil à Internet ou à des services Web}}

\newglossaryentry{internet}{
name={Internet},
description={Réseau informatique mondial accessible au public}}

\newglossaryentry{protocole}{
name={Protocole},
description={Spécification de plusieurs règles pour un type de communication particulier}}

\newglossaryentry{norme}{
name={Normes 802.11},
description={Ensemble de caractéristiques concernant les réseaux sans-fil}}

\newglossaryentry{WPA}{
name={WPA},
description={\emph{WiFi Protected Access} - Accès protégé au WiFi, un encryptage d'un réseau sans-fil }}

\newglossaryentry{lan}{
name={LAN},
description={\emph{Local Area Network} - Réseau local, réseau informatique qui relie des équipements par connexion filaire dans une zone limitée }}

\newglossaryentry{wlan}{
name={WLAN},
description={\emph{Wireless LAN} - Réseau local sans-fil, réseau informatique qui relie des équipements par ondes radios dans une zone limitée}}

\newglossaryentry{hack}{
name={Hack},
description={Accès indu à un système informatique}}

\newglossaryentry{crack}{
name={Crack},
description={Soustraction de données informatiques après avoir passé des protections mises en place}}

\newglossaryentry{ethical}{
name={Ethical hacking},
description={\emph{Hacking éthique} - Réalisation de tests d'intrusion et d'autres méthodes de test afin d'assurer la sécurité des systèmes d'information, ou dans un but éducatif}}

\newglossaryentry{mac}{
name={Adresse MAC},
description={Adresse \emph{Media Access Control} - Adresse de contrôle d'accès au support, unique au monde et aussi nommée adresse physique}}

\newglossaryentry{ip}{
name={IP},
description={\emph{Internet Protocol} - Famille de protocoles de communication de réseaux informatiques conçus pour être utilisés avec Internet}}

\newglossaryentry{ARP}{
name={ARP},
description={\emph{Address Resolution Protocol} - Protocole de résolution d'adresses, effectue la traduction d'une adresse IP en une adresse MAC}}

\newglossaryentry{paquet}{
name={Paquet},
description={Entité de transmission de données dans un réseau informatique}}

\newglossaryentry{ap}{
name={AP},
description={\emph{Access Point} - Point d'accès}}

\newglossaryentry{mode}{
name={Mode infrastructure},
description={Mode de fonctionnement d'un réseau qui relie des équipements entre eux en passant par un point d'accès}}

\newglossaryentry{mode2}{
name={Mode ad hoc},
description={Mode de fonctionnement d'un réseau qui relie des équipements directement entre eux}}

\newglossaryentry{mode3}{
name={Mode moniteur},
description={Mode de fonctionnement utilisé par un ordinateur équipé d'une carte WiFi, permet l'écoute et la capture de paquets sans connexion à un réseau }}

\newglossaryentry{carte}{
name={Carte réseau},
description={Équipement informatique assurant l'interface entre la machine sur laquelle la carte est montée et les autres équipements du réseau sur lequel elle est connectée}}

\newglossaryentry{broadcast}{
name={Broadcast},
description={Diffusion de données à tous les utilisateurs d'un réseau}}

\newglossaryentry{multicast}{
name={Multicast},
description={Diffusion de données à plusieurs utilisateurs définis d'un réseau}}

\newglossaryentry{unicast}{
name={Unicast},
description={Diffusion de données à un utilisateur d'un réseau}}

\newglossaryentry{trafic}{
name={Trafic},
description={Circulation de données sur un réseau informatique}}

\newglossaryentry{station}{
name={Station},
description={Équipement informatique connecté à un réseau, synonyme de client}}

\newglossaryentry{client}{
name={Client},
description={Équipement informatique connecté à un réseau, synonyme de station}}

\newglossaryentry{BSS}{
name={BSS},
description={\emph{Basic Service Set} - Ensemble de services de base}}

\newglossaryentry{bssid}{
name={BSSID},
description={Élément d'un BSS identifié par son ID}}

\newglossaryentry{id}{
name={ID},
description={Acronyme de \emph{Identity}, anglais pour identité}}

\newglossaryentry{ess}{
name={ESS},
description={\emph{Extended Service Set} - Ensemble de services étendu}}

\newglossaryentry{essid}{
name={ESSID},
description={Identifiant du réseau dans globalité donné par l'administrateur}}

\newglossaryentry{beacon}{
name={Trame balise},
description={Information régulièrement envoyée par un point d'accès sur son BSSID, ESSID et ses caractéristiques}}

\newglossaryentry{logiciel}{
name={Logiciel},
description={Programme informatique ,ensemble de données d'instructions interprétables par la machine}}

\newglossaryentry{integrite}{
name={Intégrité},
description={Affirmation que les données transmises n'ont pas été modifiées}}

\newglossaryentry{authentification}{
name={Authentification},
description={Vérification de l'identité d'un ordinateur afin d'autoriser l'accès de cette entité au réseau}}

\newglossaryentry{nonrepudiation}{
name={Non-répudiation},
description={Assurance que l'information transmise n'est pas remise en cause ni par l'expéditeur ni le destinataire}}

\newglossaryentry{xor}{
name={Opération XOR},
description={Opération ou exclusif, se comportant comme une addition binaire }}

\newglossaryentry{bits}{
name={Bit},
description={Unité ne pouvant prendre que deux valeurs binaires : 0 et 1, unité de mesure d'espace informatique comme la mémoire par exemple}}

\newglossaryentry{octet}{
name={Octet},
description={Regroupement de 8 bits}}

\newglossaryentry{nonce}{
name={Nonce},
description={Autre nom pour un IV}}

\newglossaryentry{iv}{
name={IV},
description={\emph{Initialization Vector} - Vecteur d'initialisation, bloc de 24 bits sensé être unique et placé à l'avant de la clé fixe dans un paquet}}

\newglossaryentry{crc}{
name={CRC},
description={Code de redondance cyclique, forme un résumé qui se situe après le message dans un paquet}}

\newglossaryentry{icv}{
name={ICV},
description={\emph{Integrity Check Value} - Valeur du contrôle d'intégrité, similaire au CRC, calculé après le cryptage du paquet et rajouté à la fin}}

\newglossaryentry{sniff}{
name={Sniffer},
description={Capture de paquets réseaux}}

\newglossaryentry{requestarp}{
name={Requête ARP},
description={Demande d'une machine à une autre pour déterminer l'adresse MAC à partir de l'adresse IP}}

\newglossaryentry{requestping}{
name={Requête ping},
description={Déterminent l'accessibilité d'un équipement en faisant une demande et attendant un écho}}

\newglossaryentry{fms}{
name={Attaques FMS},
description={Attaques exploitant les clés faibles du WEP, nommées suivant les initiales de leur créateurs}}

\newglossaryentry{mitm}{
name={MITM},
description={\emph{Man in the Middle} - L'homme du milieu, type d'attaque où le hacker se fait passer pour un point d'accès}}

\newglossaryentry{penetrationtest}{
name={Penetration Testing},
description={Test de sécurité d'un système informatique}}

\newglossaryentry{opensource}{
name={Open-source},
description={Littéralement code source ouvert, détermine les programmes et distributions libres d'accès et gratuites}}

\newglossaryentry{kali}{
name={Kali-Linux},
description={Distribution Linux de pentesting, suite de Backtrack}}

\newglossaryentry{backtrack}{
name={Backtrack},
description={Distribution Linux, une nouvelle version est sortie à ce jour et se nomme Kali-Linux}}

\newglossaryentry{liveboot}{
name={Live Boot},
description={Lancement direct}}

\newglossaryentry{partition}{
name={Partition},
description={Partie d'un disque dur destinée à accueillir des fichiers et données}}

\newglossaryentry{modem}{
name={Modem},
description={Périphérique servant à communiquer avec des utilisateurs par l'intermédiaire d'un réseau analogique comme une ligne téléphonique}}

\newglossaryentry{hub}{
name={Hub},
description={Équipement permettant d'inter-connecter plusieurs appareils informatiques}}

\newglossaryentry{terminal}{
name={Terminal},
description={Synonyme de console, programme servant à diriger un ordinateur par lignes de commande}}

\newglossaryentry{ligne de commande}{
name={Ligne de commande},
description={Texte rentré dans un terminal servant à diriger un ordinateur}}

\newglossaryentry{interface}{
name={Interface réseau},
description={Synonyme de carte réseau, aussi ce qui assure la connexion entre la machine et le réseau}}

\newglossaryentry{canal}{
name={Canal},
description={Média de transmission reliant la source au destinataire}}

\newglossaryentry{ascii}{
name={ASCII},
description={\emph{American Standard Code for Information Interchange} - Code américain normalisé pour l'échange d'information), norme de codage de caractère}}

\newglossaryentry{processus}{
name={Processus},
description={Tâche en exécution sur un ordinateur}}

\newglossaryentry{port}{
name={Port},
description={Entrées ouvertes par des programmes pour communiquer avec le système d'exploitation}}

\newglossaryentry{exploit}{
name={Exploit},
description={Un élément de programme permettant à un individu ou un logiciel malveillant d'exploiter une faille de sécurité informatique}}

\newglossaryentry{dos}{
name={DOS},
description={\emph{Deny Of Service} - Déni de Service, attaque ayant pour but de rendre un réseau indisponible}}

\newglossaryentry{bandepassante}{
name={Bande passante},
description={Désigne le débit binaire maximal d'un canal de communication}}

\newglossaryentry{backdoor}{
name={Backdoor},
description={Littéralement porte de derrière, fonctionnalité inconnue de la victime donnant accès au système d'exploitation au hacker}}

\newglossaryentry{weak}{
name={Weak Encryption Protocol},
description={Protocole d'encryptage faible}}

\newglossaryentry{linux}{
name={Linux},
description={Système d'opération libre}}
